109 /**
110 * Returns a handler name that will be used in an asynchronous getter call.
111 *
112 * @param  handler  The user-defined method that will be invoked when the getter data is returned.
113 *
114 * @return  The name of the function to invoke when getter data is received.
115 */
116 brightcove.api.modules.APIModule._getAsyncGetterHandler = function(handler) {
117     var newHandler = brightcove.api.modules.APIModule._getUniqueHandlerName();
118     brightcove.internal._handlers[newHandler] = function(result) {
119         handler(result);
120         delete brightcove.internal._handlers[newHandler];
121     };
122     return newHandler;
123 };

125 /**
126 * Invokes all handlers for an event. Generally this is not called by the API layer but is
127 * handled within the player code, but for some events, like errors, that occur in the API
128 * layer this method can be used.
129 *
130 * @param {string} event  The name of the event to listen for.
131 * @param {function} handler  The handler to invoke when the event fires.
132 */
133 brightcove.api.modules.APIModule.prototype._dispatchEvent = function(event) {
134     event.target = this;
135     var totalHandlers = this._handlers.length;
136     var handlers = [];
137     var handlerObject;
138     for (var i = 0; i < totalHandlers; i++) {
139         handlerObject = this._handlers[i];
140         if (handlerObject.event == event.type) {
141             handlers.push({handler:handlerObject.handler, priority:handlerObject.priority});
142         }
143     }
144     handlers.sort(
145         function(a, b) {
146             return b.priority - a.priority;
147         }
148     );
149     totalHandlers = handlers.length;
150     for (i = 0; i < totalHandlers; i++) {
151         handlers[i].handler(event);
152     }
153 };

155 /**
156 * Adds an event handler for the specified event.
157 *
158 * @param {string} event  The name of the event to listen for.
159 * @param {function} handler  The handler to invoke when the event fires.
160 * @param {number} priority  The priority of the handler (lower number is higher priority).  Default is 0. <em>optional</em>
161 */
162 brightcove.api.modules.APIModule.prototype._addEventListener = function(event, handler, priority) {
163     if (priority == undefined) { priority = 0; }
164     var newHandler = brightcove.api.modules.APIModule._getUniqueHandlerName();
165     this._handlers.push({handler:handler, bcHandler:newHandler, event:event, priority:priority});
166     var module = this;
167     brightcove.internal._handlers[newHandler] = function(event) {
168         event.target = module;
169         return handler(event);
170     };
171     if (this.experience.type == brightcove.playerType.FLASH) {
172         if (this.experience._playerURL) {
173             this._callMethod('addEventListener', ['event', event, newHandler, priority]);
174             return;
175         }
176         newHandler = "brightcove.internal._handlers." + newHandler;
177     }
178     this._callMethod("addEventListener", [event, newHandler, priority]);
179 };

181 /**
182 * Removes the event handler for the specified event.
183 *
184 * @param {string} event  The name of the event to listen for.
185 * @param {function} handler  The handler to invoke when the event fires.
186 */
187 brightcove.api.modules.APIModule.prototype._removeEventListener = function(event, handler) {
188     var num = this._handlers.length;
189     for (var i = 0; i < num; i++) {
190         if (this._handlers[i].event == event && this._handlers[i].handler == handler) {
191             var bcHandler = this._handlers[i].bcHandler;
192             this._handlers.splice(i, 1);
193             delete brightcove.internal._handlers[bcHandler];
194             break;
195         }
196     }
197     if (bcHandler == undefined) {
198         return;
199     }
200     if (this.experience.type == brightcove.playerType.FLASH) {
201         if (this.experience._playerURL) {
202             this._callMethod('removeEventListener', ['event', event, bcHandler]);
203             return;
204         }
205         bcHandler = "brightcove.internal._handlers." + bcHandler;
206     }
207     this._callMethod("removeEventListener", [event, bcHandler]);
208 };

292 /**
293 * Calls the embedded HTML player with the specified data.
294 *
295 * @param  params  The params (including the module and method) to pass to the player.
296 */
297 brightcove.api.modules.APIModule.prototype._callHTML = function(params) {
298     if (!this.experience._callback.postMessage) {
299         return null;
300     }
301     
302     var json;
303
304     // there is a bug where earlier versions of prototype screw up
305     // JSON conversion of arrays: BC-29035
306     if (window.Prototype != null && Prototype.Version != null) {
307         var version = Prototype.Version.split(".");
308         var majorVersion = parseInt(version[0], 10);
309         var minorVersion = parseInt(version[1], 10);
310         var oldPrototypeUsed = majorVersion == 1 && minorVersion < 7;
311         if (oldPrototypeUsed && window.console && !brightcove.internal._prototypeMessageSent) {
312             brightcove.internal._prototypeMessageSent = true;
313             var message = "An older version of prototype.js is being used on this page, preventing successful communication with ";
314             message += "the Brightcove player. The Brightcove player supports the use of version 1.7 or higher ";
315             message += "of the Prototype library.";
316             console.log(message);
317         }
318     }
319
320     if (window.JSON) {
321         json = window.JSON.stringify(params);
322     } else {
323         json = brightcove.internal._stringify(params);
324     }
325     
326     if (json) {
327         this.experience._callback.postMessage(json, this.experience._playerURL);   
328     }
329     
330     return null;
331 };

376 /**
377 * Calls a getter method in the embedded player. The handles both synchronous and asynchronous calls.
378 *
379 * @param  name  The name of the getter to invoke.
380 * @param  args  The arguments to pass to the player getter method.
381 *
382 * @return  For synchronous players (Flash) the result of the method call, otherwise null.
383 */
384 brightcove.api.modules.APIModule.prototype._callGetterMethod = function(name, args) {
385     if (args.length && typeof args[0] == "function") {
386         var handler = args.shift(),
387             handlerName;
388         if (this.experience.type == brightcove.playerType.FLASH) {
389             if (this.experience._playerURL) {
390                 handlerName = brightcove.api.modules.APIModule._getAsyncGetterHandler(handler);
391                 args.unshift(handlerName);
392                 args.unshift("getter");
393                 this._callMethod(name, args);
394             } else {
395                 var result = this._callMethod(name, args);
396                 // IE won't allow something simple like:
397                 // setTimeout(handler, 1, result);
398                 // so...
399                 setTimeout(
400                     function() {
401                         handler(result);
402                     },
403                     1
404                 );
405             }
406         } else {
407             handlerName = brightcove.api.modules.APIModule._getAsyncGetterHandler(handler);
408             args.unshift(handlerName);
409             this._callMethod(name + "Async", args);
410         }
411     } else {
412         throw "getter call must include callback function";
413     }
414 };

416 /**
417 * Class to manage API interaction with embedded Brightcove player. This constructor should not be called directly
418 * by the developer, but is handled within the Brightcove code. To obtain a reference call:
419  * @example
420  * var experience = brightcove.api.getExperience(id);
421 * @class
422 * @property {string} type The type of player being viewed, "flash" or "html".
423 */
424 /*
425 * @param {function} callback  The callback in the player to invoke for communication (for Flash). <em>optional</em>
426 * @param {string} id  The id of the embedded player. <em>optional</em>
427 */
428 brightcove.api.BrightcoveExperience = function() {
429     var callback = arguments[0];
430     var url = arguments[2];
431     this.id = arguments[1];
432     if (callback == null) {
433         this.type = brightcove.playerType.HTML;
434         this._playerURL = url;
435         this._callback = brightcove.experiences[this.id].contentWindow;
436     } else {
437         this.type = brightcove.playerType.FLASH;
438         this._callback = callback;
439         if (url) {
440             // this will signify that the flash player is inside an iFrame
441             this._playerURL = url;
442         }
443     }
444     this._modules = {};
445 };

1252 /**
1253 * Asynchronously load the DFXP file and associate with a video ID. If a video ID isn't given,
1254 * the DFXP file is loaded for the current video. After the DFXP file is loaded
1255 * and parsed, the captions will be displayed whenever the given video is played.
1256 * Events are dispatched for load error or success.
1257 * <br><br>
1258 * DFXP files must be hosted on the same domain as the containing page or must be delivered with an
1259 * appropriate Access-Control-Allow-Origin header.
1260 * <br><br>
1261 * Currently the CaptionsModule supports a subset of the DFXP standard. Supported attributes include
1262 * xml:lang, begin, and end.
1263 *
1264 * @param {string} data Either the URL for the DFXP file or a string representation of the DFXP file
1265 *                           itself.  In plugins the string representation is required.
1266 * @param {number} videoID (optional) The video ID to associate the captions with. If a video ID
1267 *                           is not provided, the DFXP file is loaded for the current video.
1268 *
1269 * @example
1270 * var APIModules = brightcove.api.modules.APIModules;
1271 * var CaptionsEvent = brightcove.api.CaptionsEvent;
1272 * var captionsModule = experience.getModule(APIModules.CAPTIONS);
1273 * captionsModule.addEventListener(CaptionsEvent.DFXP_LOAD_SUCCESS, loadHandler);
1274 * captionsModule.addEventListener(CaptionsEvent.DFXP_LOAD_ERROR, errorHandler);
1275 * captionsModule.loadDFXP("http://domain/captions.dfxp");
1276 */
1277 brightcove.api.modules.CaptionsModule.prototype.loadDFXP = function(data, videoID) {
1278     var 
1279         module = this,
1280         request,
1281         dispatchError;
1282
1283     // for HTML, we load in the parent document, not within the iframe
1284     if (this.experience.type == brightcove.playerType.HTML) {
1285             dispatchError = function(error) {
1286             if (window.console) {
1287                 console.log(error);
1288             }
1289             module._dispatchEvent(
1290                 {
1291                 type:brightcove.api.events.CaptionsEvent.DFXP_LOAD_ERROR,
1292                 url:data,
1293                 error:error
1294                 }
1295             );
1296         };
1297         if (data.trim().indexOf('<') === 0) {
1298             // data is literal XML, call loadDFXP directly
1299             module._callMethod("loadDFXP", [data, videoID]);
1300             return undefined;
1301         }
1302         if (window.XMLHttpRequest) {
1303             // is there a danger of garbage collection here?
1304             request = new XMLHttpRequest();
1305             request.onreadystatechange = function() {
1306                 if (request.readyState === 4) {
1307                     try {
1308                         if (request.status === 200) {
1309                             module._callMethod("parseDFXP", [data, request.responseText, videoID]);
1310                         } else {
1311                             dispatchError("Error loading DFXP file. " +
1312                                           request.statusText +
1313                                           " HTTP status: " +
1314                                           request.status);
1315                         }
1316                     } catch (e) {
1317                         dispatchError(e.description);
1318                     }
1319                 }
1320             };
1321             request.open("GET", data, true);
1322             request.send(null);
1323         } else {
1324             dispatchError("Browser not supported by Brightcove player for loading DFXP file.");

1946 /**
1947 * Adds an event handler for the specified event.
1948 *
1949 * @param {string} event  The name of the event to listen for.
1950 * @param {function} handler  The handler to invoke when the event fires.
1951 * @param {number} priority  The priority of the handler (lower number is higher priority).  Default is 0. <em>optional</em>
1952 *
1953 * @example
1954 *
1955 * var APIModules = brightcove.api.modules.APIModules;
1956 * var videoPlayer = experience.getModule(APIModules.VIDEO_PLAYER);

1958 *       brightcove.api.events.MediaEvent.COMPLETE,
1959 *       myCompleteHandler);
1960 */
1961 brightcove.api.modules.VideoPlayerModule.prototype.addEventListener = function(event, handler, priority) {
1962     // make sure not to add the same handler twice
1963     this.removeEventListener(event, handler);
1964     var wrapper;
1965     if (event == brightcove.api.events.CuePointEvent.CUE) {
1966         wrapper = function(pEvent) {
1967             pEvent.cuePoint = new brightcove.api.data.CuePoint(pEvent.cuePoint);
1968             handler(pEvent);
1969         };
1970         this._handlerWrappers.push({event:event, handler:handler, wrapper:wrapper});
1971     }
1972     this._addEventListener(event, wrapper || handler, priority);
1973 };

1976 * Removes the event handler for the specified event.
1977 *
1978 * @param {string} event  The name of the event to listen for.
1979 * @param {function} handler  The handler to invoke when the event fires.
1980 *
1981 * @example
1982 * var APIModules = brightcove.api.modules.APIModules;

1985 *       brightcove.api.events.MediaEvent.PROGRESS,
1986 *       myProgressHandler);

1989     if (event == brightcove.api.events.CuePointEvent.CUE) {
1990         var wrapper;
1991         var num = this._handlerWrappers.length;
1992         for (var i = 0; i < num; i++) {
1993             if (this._handlerWrappers[i].event == event && this._handlerWrappers[i].handler == handler) {
1994                 wrapper = this._handlerWrappers[i].wrapper;

1998         }
1999     }
